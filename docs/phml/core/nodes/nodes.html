 21class Point:
 22    """Represents one place in a source file.
 23
 24    The line field (1-indexed integer) represents a line in a source file. The column field
 25    (1-indexed integer) represents a column in a source file. The offset field (0-indexed integer)
 26    represents a character in a source file.
 27    """
 28
 29    def __init__(self, line: int, column: int, offset: Optional[int] = None):
 30        if line is None or line < 0:
 31            raise IndexError(f"Point.line must be >= 0 but was {line}")
 32
 33        self.line = line
 34
 35        if column is None or column < 0:
 36            raise IndexError(f"Point.column must be >= 0 but was {column}")
 37
 38        self.column = column
 39
 40        if offset is not None and offset < 0:
 41            raise IndexError(f"Point.offset must be >= 0 or None but was {line}")
 42
 43        self.offset = offset
 44
 45    def __eq__(self, obj) -> bool:
 46        return bool(
 47            obj is not None
 48            and isinstance(obj, self.__class__)
 49            and self.line == obj.line
 50            and self.column == obj.column
 51        )
 52
 53    def __repr__(self) -> str:
 54        return f"point(line: {self.line}, column: {self.column}, offset: {self.offset})"
 55
 56    def __str__(self) -> str:
 57        return f"{self.line}:{self.column}"

 59class Position:
 60    """Position represents the location of a node in a source file.
 61
 62    The `start` field of `Position` represents the place of the first character
 63    of the parsed source region. The `end` field of Position represents the place
 64    of the first character after the parsed source region, whether it exists or not.
 65    The value of the `start` and `end` fields implement the `Point` interface.
 66
 67    The `indent` field of `Position` represents the start column at each index
 68    (plus start line) in the source region, for elements that span multiple lines.
 69
 70    If the syntactic unit represented by a node is not present in the source file at
 71    the time of parsing, the node is said to be `generated` and it must not have positional
 72    information.
 73    """
 74
 75    @overload
 76    def __init__(
 77        self,
 78        start: tuple[int, int, int | None],
 79        end: tuple[int, int, int | None],
 80        indent: Optional[int] = None,
 81    ):
 82        """
 83        Args:
 84            start (tuple[int, int, int  |  None]): Tuple representing the line, column, and optional
 85            offset of the start point.
 86            end (tuple[int, int, int  |  None]): Tuple representing the line, column, and optional
 87            offset of the end point.
 88            indent (Optional[int], optional): The indent amount for the start of the position.
 89        """
 90        ...
 91
 92    def __init__(self, start: Point, end: Point, indent: Optional[int] = None):
 93        """
 94        Args:
 95            start (Point): Starting point of the position.
 96            end (Point): End point of the position.
 97            indent (int | None): The indent amount for the start of the position.
 98        """
 99
100        self.start = (
101            Point(start[0], start[1], start[2] if len(start) == 3 else None)
102            if isinstance(start, tuple)
103            else start
104        )
105        self.end = (
106            Point(end[0], end[1], end[2] if len(end) == 3 else None)
107            if isinstance(end, tuple)
108            else end
109        )
110
111        if indent is not None and indent < 0:
112            raise IndexError(f"Position.indent value must be >= 0 or None but was {indent}")
113
114        self.indent = indent
115
116    def __eq__(self, obj) -> bool:
117        return bool(
118            obj is not None
119            and isinstance(obj, Position)
120            and self.start == obj.start
121            and self.end == obj.end
122        )
123
124    def as_dict(self) -> dict:
125        """Convert the position object to a dict."""
126        return {
127            "start": {
128                "line": self.start.line,
129                "column": self.start.column,
130                "offset": self.start.offset,
131            },
132            "end": {"line": self.end.line, "column": self.end.column, "offset": self.end.offset},
133            "indent": self.indent,
134        }
135
136    def __repr__(self) -> str:
137        indent = f" ~ {self.indent}" if self.indent is not None else ""
138        return f"<{self.start}-{self.end}{indent}>"
139
140    def __str__(self) -> str:
141        return repr(self)

169class Parent(Node):  # pylint: disable=too-few-public-methods
170    """Parent (UnistParent) represents a node in hast containing other nodes (said to be children).
171
172    Its content is limited to only other hast content.
173    """
174
175    def __init__(self, position: Optional[Position] = None, children: Optional[list] = None):
176        super().__init__(position)
177
178        if children is not None:
179            for child in children:
180                if hasattr(child, "type") and child.type in [
181                    "element",
182                    "text",
183                    "doctype",
184                    "root",
185                    "comment",
186                ]:
187                    child.parent = self
188
189        self.children: list[Element | DocType | Comment | Text] = children or []
190
191    def append(self, node: NODE):
192        """Add a node to the nested children of the current parent node."""
193        node.parent = self
194        self.children.append(node)
195        
196    def extend(self, nodes: list[NODE]):
197        """Add a node to the nested children of the current parent node."""
198        for node in nodes:
199            self.append(node)
200
201    def insert(self, index: int, node: NODE):
202        """Insert a node into a specific position in the current parent node's children."""
203        node.parent = self
204        self.children.insert(index, node)
205
206    def remove(self, node: NODE):
207        """Remove a specific node from the current parent node's children."""
208        self.children.remove(node)

501class Comment(Literal):
502    """Comment (Literal) represents a Comment ([DOM]).
503
504    Example:
505    ```html
506    <!--Charlie-->
507    ```
508    """
509
510    def stringify(self, indent: int = 0) -> str:
511        """Build indented html string of html comment.
512
513        Returns:
514            str: Built html of comment
515        """
516        lines = [line for line in self.value.split("\n") if line.strip() != ""]
517        if len(lines) > 1:
518            start = f"{' ' * indent}<!--{lines[0].rstrip()}"
519            end = f"{' ' * indent}{lines[-1].lstrip()}-->"
520            for i in range(1, len(lines) - 1):
521                lines[i] = (' ' * indent) + lines[i].strip()
522            lines = [start, *lines[1:-1], end]
523            return "\n".join(lines)
524        return ' ' * indent + f"<!--{self.value}-->"
525
526    def __repr__(self) -> str:
527        return f"literal.comment(value: {self.value})"
